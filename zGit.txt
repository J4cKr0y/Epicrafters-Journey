# Créer et basculer sur une nouvelle branche
git checkout -b nouvelle-branche

# Faire des modifications, puis ajouter et committer les changements
git add .
git commit -m "Ajout de nouvelles fonctionnalités"

# Pousser la nouvelle branche vers le dépôt distant
git push origin nouvelle-branche

git stash

# Basculer sur la branche principale et la mettre à jour
git checkout master
git pull origin master

# Fusionner la nouvelle branche avec la branche principale
git merge nouvelle-branche

# Pousser les modifications fusionnées vers le dépôt distant
git push origin master
